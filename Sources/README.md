# Commands:

    git clone https://github.com/demining/CryptoDeepTools.git

    cd CryptoDeepTools/11QianshiBTC/
    
    sudo apt-get install g++ -y
    
    sudo apt-get install libgmp3-dev libmpfr-dev -y
    
    make
    
    ./qianshiBTC 40
